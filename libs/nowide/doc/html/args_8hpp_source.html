86  {
87  int wargc;
88  wchar_t **wargv = CommandLineToArgvW(GetCommandLineW(),&wargc);
89  if(!wargv) {
90  argc = 0;
91  static char *dummy = 0;
92  argv = &dummy;
93  return;
94  }
95  try{

97  arg_values_.resize(wargc);
98  for(int i=0;i<wargc;i++) {
99  if(!arg_values_[i].convert(wargv[i])) {
